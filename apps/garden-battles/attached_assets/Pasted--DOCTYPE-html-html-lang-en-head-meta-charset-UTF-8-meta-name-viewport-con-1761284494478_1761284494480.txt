                // Tree and particle animation on canvas
                const canvas = document.getElementById('tree-canvas');
                const ctx = canvas.getContext('2d');
                if (!ctx) {
                    console.error('Canvas context not supported');
                    return;
                }
                canvas.width = window.innerWidth;
                canvas.height = window.innerHeight;
                window.addEventListener('resize', () => {
                    canvas.width = window.innerWidth;
                    canvas.height = window.innerHeight;
                });
                class Particle {
                    constructor(x, y) {
                        this.x = x;
                        this.y = y;
                        this.size = Math.random() * 5 + 5;
                        this.vx = (Math.random() - 0.5) * 6;
                        this.vy = (Math.random() - 0.5) * 6;
                        this.opacity = 1;
                        this.style = document.documentElement.style;
                        this.style.setProperty('--tx', `${this.vx * 50}px`);
                        this.style.setProperty('--ty', `${this.vy * 50}px`);
                    }
                    draw() {
                        ctx.fillStyle = `rgba(52, 211, 153, ${this.opacity})`;
                        ctx.beginPath();
                        ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                        ctx.fill();
                    }
                    update() {
                        this.x += this.vx;
                        this.y += this.vy;
                        this.opacity -= 0.015;
                        this.size *= 0.98;
                    }
                }
                class Branch {
                    constructor(x, y, angle, length, depth, treeId) {
                        this.x = x;
                        this.y = y;
                        this.angle = angle;
                        this.length = length;
                        this.depth = depth;
                        this.treeId = treeId;
                        this.growth = 0;
                        this.maxLength = length;
                        this.leaves = [];
                        this.particles = [];
                        this.isBursting = false;
                        this.growthComplete = false;
                    }
                    grow() {
                        if (this.growth < this.maxLength) {
                            this.growth += 0.1; // Very slow growth
                            if (this.growth >= this.maxLength && !this.growthComplete) {
                                this.growthComplete = true;
                                if (this.depth > 0) {
                                    const numSplits = Math.floor(Math.random() * 3) + 2; // 2–4 splits
                                    for (let i = 0; i < numSplits; i++) {
                                        const newAngle = this.angle + (Math.random() - 0.5) * Math.PI / 3;
                                        const newLength = this.length * (0.5 + Math.random() * 0.3);
                                        const newBranch = new Branch(
                                            this.x + Math.cos(this.angle) * this.length,
                                            this.y + Math.sin(this.angle) * this.length,
                                            newAngle,
                                            newLength,
                                            this.depth - 1,
                                            this.treeId
                                        );
                                        trees.push(newBranch);
                                    }
                                }
                                if (this.depth === 0 && this.leaves.length === 0) {
                                    for (let i = 0; i < 3; i++) {
                                        this.leaves.push({
                                            x: this.x + Math.cos(this.angle) * this.growth,
                                            y: this.y + Math.sin(this.angle) * this.growth,
                                            size: Math.random() * 8 + 4
                                        });
                                    }
                                }
                            }
                        } else if (this.growthComplete && this.depth === 0 && !this.isBursting) {
                            if (Math.random() < 0.005) { // Slow transition to bursting
                                this.isBursting = true;
                                for (let i = 0; i < 8; i++) {
                                    this.particles.push(new Particle(
                                        this.x + Math.cos(this.angle) * this.growth,
                                        this.y + Math.sin(this.angle) * this.growth
                                    ));
                                }
                            }
                        }
                    }
                    draw() {
                        if (this.isBursting) {
                            this.particles.forEach(particle => {
                                particle.update();
                                particle.draw();
                            });
                            return;
                        }
                        // Draw branch
                        ctx.beginPath();
                        ctx.moveTo(this.x, this.y);
                        const endX = this.x + Math.cos(this.angle) * this.growth;
                        const endY = this.y + Math.sin(this.angle) * this.growth;
                        ctx.lineTo(endX, endY);
                        ctx.strokeStyle = 'var(--tree-brown)';
                        ctx.lineWidth = this.depth + 1;
                        ctx.stroke();
                        // Draw leaves
                        if (this.growthComplete && this.depth === 0) {
                            this.leaves.forEach(leaf => {
                                ctx.fillStyle = 'var(--green-light)';
                                ctx.beginPath();
                                ctx.arc(leaf.x, leaf.y, leaf.size, 0, Math.PI * 2);
                                ctx.fill();
                            });
                        }
                    }
                }
                const trees = [];
                const maxTrees = 5;
                let treeIdCounter = 0;
                function spawnTree() {
                    if (trees.length >= maxTrees) return;
                    const x = Math.random() * canvas.width;
                    const y = canvas.height; // Start at bottom
                    const angle = -Math.PI / 2 + (Math.random() - 0.5) * Math.PI / 6; // Narrow upward angle
                    const length = 50 + Math.random() * 100;
                    trees.push(new Branch(x, y, angle, length, 2, treeIdCounter++));
                }
                let lastFrame = 0;
                function animate() {
                    const now = performance.now();
                    if (now - lastFrame < 16) return requestAnimationFrame(animate);
                    lastFrame = now;
                    ctx.clearRect(0, 0, canvas.width, canvas.height);
                    trees.forEach(tree => {
                        tree.grow();
                        tree.draw();
                    });
                    // Track bursting branches and remove trees when all final branches have burst
                    const treeBurstingStatus = new Map();
                    trees.forEach(tree => {
                        if (tree.depth === 0 && tree.isBursting && tree.particles.every(p => p.opacity <= 0)) {
                            treeBurstingStatus.set(tree.treeId, (treeBurstingStatus.get(tree.treeId) || 0) + 1);
                        }
                    });
                    const treeIdsToRemove = new Set();
                    treeBurstingStatus.forEach((burstCount, treeId) => {
                        const finalBranches = trees.filter(t => t.treeId === treeId && t.depth === 0).length;
                        if (burstCount === finalBranches && finalBranches > 0) {
                            treeIdsToRemove.add(treeId);
                        }
                    });
                    for (let i = trees.length - 1; i >= 0; i--) {
                        if (treeIdsToRemove.has(trees[i].treeId)) {
                            trees.splice(i, 1);
                        }
                    }
                    // Spawn new trees only if there are fewer than maxTrees and no trees are in the process of bursting
                    if (trees.length < maxTrees && trees.every(tree => !tree.isBursting) && Math.random() < 0.02) {
                        spawnTree();
                    }
                    requestAnimationFrame(animate);
                }
                // Start with 3 trees
                for (let i = 0; i < 3; i++) {
                    spawnTree();
                }
                animate();
            });
        } catch (error) {
            console.error('Script error:', error);
            alert('An error occurred. Please refresh.');
